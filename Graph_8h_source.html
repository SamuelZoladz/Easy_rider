281 static bool segmentCrosses(const std::pair<int, int> &f,
282 const std::pair<int, int> &t,
283 const std::pair<int, int> &q1,
284 const std::pair<int, int> &q2) {
285 if (q1 == f || q1 == t || q2 == f || q2 == t)
286 return false;
287
288 long long o1 = orient(f, t, q1);

290 long long o3 = orient(q1, q2, f);
291 long long o4 = orient(q1, q2, t);
292
293 if (((o1 > 0 && o2 < 0) || (o1 < 0 && o2 > 0)) &&
294 ((o3 > 0 && o4 < 0) || (o3 < 0 && o4 > 0)))
295 return true;
296
297 if (o1 == 0 && onSegment(f, t, q1))
298 return true;
299 if (o2 == 0 && onSegment(f, t, q2))
300 return true;
301 if (o3 == 0 && onSegment(q1, q2, f))
302 return true;
303 if (o4 == 0 && onSegment(q1, q2, t))
304 return true;
